📦 Variables & Data Types

Variables (let, const, var)

Variable ek container hai jisme data store hota hai.

// let — reassign possible, block scope

// const — reassign NOT possible, block scope

// pi = 3.14; // ❌ Error!

// var — old way, function scope (avoid karo)
var oldWay = "deprecated";

➕ Operators

Arithmetic Operators

let a = 10, b = 3;
console.log(a + b);  // 13
console.log(a - b);  // 7
console.log(a * b);  // 30
console.log(a / b);  // 3.333...
console.log(a % b);  // 1 (remainder)
console.log(a ** b); // 1000 (power: 10^3)

// Increment / Decrement
let x = 5;
console.log(x++); // 5 (post-increment)
console.log(++x); // 7 (pre-increment)

Comparison Operators

console.log(5 == "5");   // true (loose equality, type conversion)
console.log(5 === "5");  // false (strict equality, type + value)
console.log(5 != "5");   // false
console.log(5 !== "5");  // true
console.log(10 > 5);     // true
console.log(10 <= 10);   // true
Always use === and !== instead of == and != to avoid unexpected type conversion bugs.

Logical Operators

let age = 20, hasID = true;
console.log(age > 18 && hasID);  // true (AND)
console.log(age < 18 || hasID);  // true (OR)
console.log(!hasID);             // false (NOT)

// Short-circuit evaluation
let user = null;
let name = user || "Guest";  // "Guest" (falsy value fallback)

Ternary Operator

let age = 20;
let status = age >= 18 ? "Adult" : "Minor";
console.log(status); // "Adult"

🔄 Control Flow

If-Else

let marks = 85;

if (marks >= 90) {
    console.log("Grade A+");
} else if (marks >= 80) {
    console.log("Grade A");
} else if (marks >= 70) {
    console.log("Grade B");
} else if (marks >= 60) {
    console.log("Grade C");
} else {
    console.log("Fail");
}

Switch Case

let day = 3;
let dayName;

switch (day) {
    case 1: dayName = "Monday"; break;
    case 2: dayName = "Tuesday"; break;
    case 3: dayName = "Wednesday"; break;
    case 4: dayName = "Thursday"; break;
    case 5: dayName = "Friday"; break;
    case 6: dayName = "Saturday"; break;
    case 7: dayName = "Sunday"; break;
    default: dayName = "Invalid day";
}
console.log(dayName); // "Wednesday"

Loops

// For loop
for (let i = 0; i < 5; i++) {
    console.log(i); // 0, 1, 2, 3, 4
}

// While loop
let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}

// Do-While loop (at least once chalega)
let j = 0;
do {
    console.log(j);
    j++;
} while (j < 5);

// For...of (arrays ke liye)
let fruits = ["Apple", "Banana", "Mango"];
for (let fruit of fruits) {
    console.log(fruit);
}

// For...in (objects ke liye)
let person = { name: "Akarshit", age: 20 };
for (let key in person) {
    console.log(key + ": " + person[key]);
}

Loop Control

for (let i = 0; i < 10; i++) {
    if (i === 3) continue; // skip 3
    if (i === 7) break;    // stop at 7
    console.log(i); // 0,1,2,4,5,6
}

🔧 Functions

Function Declaration

function greet(name) {
    return "Hello, " + name + "!";
}
console.log(greet("Akarshit")); // "Hello, Akarshit!"

Function Expression

const add = function(a, b) {
    return a + b;
};
console.log(add(5, 3)); // 8

Arrow Functions (ES6)

// Single parameter (parentheses optional)
const square = x => x * x;
console.log(square(5)); // 25

// Multiple parameters
const multiply = (a, b) => a * b;
console.log(multiply(4, 5)); // 20

// Multiple statements
const greet = (name) => {
    let msg = "Hello, " + name;
    return msg;
};

Default Parameters

function greet(name = "Guest") {
    return "Hello, " + name;
}
console.log(greet());        // "Hello, Guest"
console.log(greet("Raj"));   // "Hello, Raj"

Rest Parameters

function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}
console.log(sum(1, 2, 3, 4, 5)); // 15

Higher-Order Functions

// Function returning function
function createMultiplier(factor) {
    return function(number) {
        return number * factor;
    };
}
const double = createMultiplier(2);
console.log(double(5)); // 10

// Callback function
function processData(data, callback) {
    let result = data.toUpperCase();
    callback(result);
}
processData("hello", (res) => console.log(res)); // "HELLO"

📋 Arrays

Creating Arrays

let fruits = ["Apple", "Banana", "Mango"];
let numbers = new Array(1, 2, 3, 4, 5);
let mixed = [1, "hello", true, {a: 1}, [1, 2]];

Array Methods

let arr = [1, 2, 3, 4, 5];

// push/pop (end se)
arr.push(6);        // [1,2,3,4,5,6]
arr.pop();          // [1,2,3,4,5]

// shift/unshift (start se)
arr.shift();        // [2,3,4,5] (remove first)
arr.unshift(0);     // [0,2,3,4,5] (add first)

// splice (add/remove anywhere)
arr.splice(2, 1);        // index 2 se 1 element remove
arr.splice(2, 0, "a");   // index 2 par "a" insert

// slice (copy portion)
let sliced = arr.slice(1, 3); // elements 1 to 2

// indexOf, includes
console.log(arr.indexOf(3));   // index of 3
console.log(arr.includes(5));  // true/false

Iteration Methods

let nums = [1, 2, 3, 4, 5];

// forEach
nums.forEach((num, index) => {
    console.log(index + ": " + num);
});

// map (transform each element)
let doubled = nums.map(num => num * 2);
// [2, 4, 6, 8, 10]

// filter (select elements)
let evens = nums.filter(num => num % 2 === 0);
// [2, 4]

// reduce (accumulate)
let sum = nums.reduce((total, num) => total + num, 0);
// 15

// find (first matching element)
let firstEven = nums.find(num => num % 2 === 0);
// 2

// sort
let unsorted = [3, 1, 4, 1, 5];
unsorted.sort((a, b) => a - b); // [1, 1, 3, 4, 5]

// reverse
unsorted.reverse(); // [5, 4, 3, 1, 1]

// every/some
console.log(nums.every(n => n > 0)); // true
console.log(nums.some(n => n > 4));  // true

2D Arrays

let matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
];
console.log(matrix[1][2]); // 6 (row 1, col 2)

// Nested loop
for (let i = 0; i < matrix.length; i++) {
    for (let j = 0; j < matrix[i].length; j++) {
        console.log(matrix[i][j]);
    }
}

📦 Objects

Creating Objects

let person = {
    firstName: "Akarshit",
    lastName: "Raj",
    age: 20,
    isStudent: true,
    hobbies: ["Coding", "Teaching", "Gaming"],
    address: {
        city: "Sitapur",
        state: "UP"
    },
    // Method
    fullName: function() {
        return this.firstName + " " + this.lastName;
    }
};

console.log(person.firstName);      // "Akarshit"
console.log(person["lastName"]);    // "Raj"
console.log(person.fullName());     // "Akarshit Raj"
console.log(person.address.city);   // "Sitapur"

Object Methods (ES6+)

// Object.keys, values, entries
console.log(Object.keys(person));   // ["firstName", "lastName", ...]
console.log(Object.values(person)); // ["Akarshit", "Raj", 20, ...]
console.log(Object.entries(person)); // [["firstName", "Akarshit"], ...]

// Object.assign (merge objects)
let defaults = { theme: "dark", lang: "en" };
let user = { lang: "hi" };
let settings = Object.assign({}, defaults, user);
// { theme: "dark", lang: "hi" }

// Spread operator
let clone = { ...person };
let merged = { ...defaults, ...user };

Destructuring

// Object destructuring
let { firstName, age } = person;
console.log(firstName, age);

// With alias
let { firstName: fname } = person;

// Nested destructuring
let { address: { city } } = person;
console.log(city); // "Sitapur"

// Array destructuring
let [a, b, c] = [1, 2, 3];
let [first, ...rest] = [1, 2, 3, 4, 5]; // first=1, rest=[2,3,4,5]

JSON

// Object to JSON string
let jsonStr = JSON.stringify(person);

// JSON string to Object
let parsed = JSON.parse(jsonStr);

🌐 DOM Manipulation

DOM (Document Object Model) HTML page ka tree structure hai. JavaScript se isko modify kar sakte hain.

Selecting Elements

// By ID
let element = document.getElementById("myId");

// By Class Name (returns HTMLCollection)
let elements = document.getElementsByClassName("myClass");

// By Tag Name
let divs = document.getElementsByTagName("div");

// CSS Selector (modern way)
let first = document.querySelector(".myClass");
let all = document.querySelectorAll(".myClass"); // NodeList

Modifying Elements

let el = document.getElementById("demo");

// Content
el.innerHTML = "<b>Bold Text</b>";  // HTML parse hoga
el.textContent = "Plain Text";      // Raw text

// Attributes
el.setAttribute("class", "highlight");
el.getAttribute("class");
el.removeAttribute("class");

// Style
el.style.color = "red";
el.style.backgroundColor = "yellow";
el.style.fontSize = "20px";

// Classes
el.classList.add("active");
el.classList.remove("active");
el.classList.toggle("active");
el.classList.contains("active"); // true/false

Creating & Removing Elements

// Create new element
let newDiv = document.createElement("div");
newDiv.textContent = "I am new!";
newDiv.className = "box";

// Add to DOM
document.body.appendChild(newDiv);

// Insert before another element
let parent = document.getElementById("container");
let reference = document.getElementById("ref");
parent.insertBefore(newDiv, reference);

// Modern: insertAdjacentHTML
parent.insertAdjacentHTML("beforeend", "<p>New paragraph</p>");

// Remove element
let old = document.getElementById("remove-me");
old.remove(); // Modern browsers
// OR
old.parentNode.removeChild(old);

🎯 Events

Events user interactions hain — click, hover, keypress, etc.

Event Listeners

let btn = document.getElementById("myBtn");

// Add event listener
btn.addEventListener("click", function(event) {
    console.log("Button clicked!");
    console.log(event.target); // clicked element
});

// Arrow function
btn.addEventListener("click", (e) => {
    console.log("Clicked!");
    e.preventDefault(); // default action roko
});

// Remove listener
function handler() { console.log("Hi"); }
btn.addEventListener("click", handler);
btn.removeEventListener("click", handler);

Destructuring

Spread & Rest Operators

// Spread (expand)
let arr1 = [1, 2, 3];
let arr2 = [...arr1, 4, 5]; // [1,2,3,4,5]
let obj1 = {a: 1};

    return nums.reduce((a, b) => a + b, 0);
}
let [head, ...tail] = [1, 2, 3, 4]; // head=1, tail=[2,3,4]

⏳ Async JavaScript

Callbacks

function fetchData(callback) {
    setTimeout(() => {
        callback("Data loaded!");
    }, 1000);
}
fetchData((msg) => console.log(msg));

Promises

let promise = new Promise((resolve, reject) => {
    let success = true;
    if (success) {
        resolve("Done!");
    } else {
        reject("Error!");
    }
});

promise
    .then(result => console.log(result))
    .catch(error => console.error(error))
    .finally(() => console.log("Finished"));

Promise Chaining

fetchUser(1)
    .then(user => fetchOrders(user.id))
    .then(orders => console.log(orders))
    .catch(err => console.error(err));

Async/Await

async function getUserData(userId) {
    try {
        let user = await fetchUser(userId);
        let orders = await fetchOrders(user.id);
        console.log(orders);
    } catch (error) {
        console.error("Error:", error);
    } finally {
        console.log("Request completed");
    }
}

// Parallel execution
async function loadAll() {
    let [users, posts] = await Promise.all([
        fetchUsers(),
        fetchPosts()
    ]);
}

Promise.all / Promise.race

// All promises complete hone ke baad
let results = await Promise.all([p1, p2, p3]);

// Sabse pehle complete hone wala
let winner = await Promise.race([p1, p2, p3]);

🌐 Fetch API & AJAX

GET Request

.then(res => res.json())
.then(data => console.log(data));

🛠️ Mini Projects

1. To-Do List

let tasks = [];

function addTask(task) {
    tasks.push({ text: task, done: false });
    render();
}

function toggleTask(index) {
    tasks[index].done = !tasks[index].done;
    render();
}

function render() {
    let list = document.getElementById("taskList");
    list.innerHTML = tasks.map((t, i) => `
        <li class="${t.done ? 'done' : ''}" onclick="toggleTask(${i})">
            ${t.text}
        </li>
    `).join("");
}

2. Counter App

let count = 0;

function updateDisplay() {
    document.getElementById("count").textContent = count;
}

function increment() { count++; updateDisplay(); }
function decrement() { count--; updateDisplay(); }
function reset() { count = 0; updateDisplay(); }

OOP - Classes & Objects

class Student {
    String name;
    int age;
    
    Student(String n, int a) {
        name = n;
        age = a;
    }
    
    void display() {
        System.out.println(name + " - " + age);
    }
}

public class Main {
    public static void main(String[] args) {
        Student s1 = new Student("Akarshit", 18);
        s1.display();
    }
}

Inheritance

class Animal {
    void eat() { System.out.println("Eating..."); }
}

class Dog extends Animal {
    void bark() { System.out.println("Barking..."); }
}

Dog d = new Dog();
d.eat();   // Inherited
d.bark();  // Own method

➕ C++ Operators Complete Guide

Operators special symbols hain jo variables aur values par operations perform karte hain. C++ mein 7 types ke operators hain.

1️⃣ Arithmetic Operators

int a = 10, b = 3;
cout << a + b << endl;   // 13
cout << a - b << endl;   // 7
cout << a * b << endl;   // 30
cout << a / b << endl;   // 3 (integer division)
cout << a % b << endl;   // 1 (remainder)

2️⃣ Comparison Operators

int x = 10, y = 5;
cout << (x == y) << endl;  // 0 (false)
cout << (x != y) << endl;  // 1 (true)
cout << (x > y) << endl;   // 1 (true)
cout << (x < y) << endl;   // 0 (false)
cout << (x >= y) << endl;  // 1 (true)
cout << (x <= y) << endl;  // 0 (false)

3️⃣ Assignment Operators

int score = 50;
score += 10;    // score = score + 10
score -= 5;     // score = score - 5
score *= 2;     // score = score * 2
score /= 2;     // score = score / 2
cout << score << endl;  // 55

4️⃣ Logical Operators

int age = 17, marks = 85;
cout << (age > 15 && marks > 80) << endl;  // 1
cout << (age > 20 || marks > 80) << endl;  // 1
cout << !(age > 20) << endl;               // 1

5️⃣ Increment / Decrement Operators

int i = 5;
cout << i++ << endl;  // 5 (phir i = 6)
cout << ++i << endl;  // 7 (pehle i = 7)

cout << i-- << endl;  // 7 (phir i = 6)
cout << --i << endl;  // 5 (pehle i = 5)

6️⃣ Bitwise Operators

int a = 5, b = 3;
cout << (a & b) << endl;   // 1 (AND)
cout << (a | b) << endl;   // 7 (OR)
cout << (a ^ b) << endl;   // 6 (XOR)
cout << (~a) << endl;      // -6 (NOT)
cout << (a << 1) << endl;  // 10 (Left shift)
cout << (a >> 1) << endl;  // 2 (Right shift)

7️⃣ Ternary Operator

int marks = 85;
string result = (marks >= 40) ? "Pass" : "Fail";
cout << result << endl;
💡 Yaad Rakhein: C++ mein / integer division karta hai agar dono integers hain. Decimal chahiye toh float ya double use karein!

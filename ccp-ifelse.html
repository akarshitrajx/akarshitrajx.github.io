C++ If-Else Complete Guide

Basic If-Else

int age = 18;
if (age >= 18) {
    cout << "Adult";
} else {
    cout << "Minor";
}

Else If

int marks = 85;
if (marks >= 90) {
    cout << "Grade A";
} else if (marks >= 80) {
    cout << "Grade B";
} else if (marks >= 70) {
    cout << "Grade C";
} else {
    cout << "Fail";
}

Nested If

int a = 10, b = 20;
if (a > 5) {
    if (b > 15) {
        cout << "Both conditions true";
    }
}

Switch Statement

int day = 3;
switch (day) {
    case 1: cout << "Monday"; break;
    case 2: cout << "Tuesday"; break;
    case 3: cout << "Wednesday"; break;
    default: cout << "Invalid";
}

Ternary Operator

int a = 10, b = 20;
int max = (a > b) ? a : b;
cout << max;  // 20

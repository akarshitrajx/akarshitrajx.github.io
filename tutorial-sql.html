SQL Complete Tutorial

What is SQL?

SQL (Structured Query Language) is used to manage databases. It helps you store, retrieve, and manipulate data.

Create Table

CREATE TABLE students (
    id INT PRIMARY KEY,
    name VARCHAR(50),
    age INT,
    grade CHAR(1)
);

INSERT Data

INSERT INTO students VALUES (1, 'Akarshit', 18, 'A');
INSERT INTO students (name, age) VALUES ('Rahul', 17);

SELECT Queries

SELECT * FROM students;
SELECT name, age FROM students WHERE grade = 'A';
SELECT * FROM students ORDER BY age DESC;
SELECT COUNT(*) FROM students;

UPDATE & DELETE

UPDATE students SET age = 19 WHERE id = 1;
DELETE FROM students WHERE id = 2;

JOIN Tables

SELECT s.name, m.subject, m.marks
FROM students s
JOIN marks m ON s.id = m.student_id;

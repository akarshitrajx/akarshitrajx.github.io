🚀 Introduction

SQL (Structured Query Language) databases se baat karne ki language hai. Har bada website, app, aur software database use karta hai — SQL us data ko manage, query, aur manipulate karne ka standard way hai.

Kya seekhenge:

  • Tables create karna, data insert karna
  • SELECT, WHERE, ORDER BY queries
  • JOINs — data multiple tables se connect karna
  • Aggregate functions, GROUP BY, HAVING
  • Constraints, Indexes, Subqueries
  • Real-world database design
Popular Databases: MySQL, PostgreSQL, SQLite, SQL Server, Oracle. Syntax almost same hota hai sabmein.

🏗️ CREATE TABLE

Table database mein data store karne ka structure hai. Rows (records) aur Columns (fields) se bana hota hai.

Basic Syntax

CREATE TABLE table_name (
    column1 datatype constraints,
    column2 datatype constraints,
    ...
);

Example: Students Table

CREATE TABLE students (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE,
    age INT,
    grade CHAR(2),
    marks DECIMAL(5,2),
    enrollment_date DATE DEFAULT CURRENT_DATE,
    is_active BOOLEAN DEFAULT TRUE
);

Example: Multiple Tables

CREATE TABLE departments (
    dept_id INT PRIMARY KEY AUTO_INCREMENT,
    dept_name VARCHAR(50) NOT NULL,
    location VARCHAR(50)
);

CREATE TABLE employees (
    emp_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    salary DECIMAL(10,2),
    dept_id INT,
    hire_date DATE,
    FOREIGN KEY (dept_id) REFERENCES departments(dept_id)
);

🔍 SELECT Queries

Basic SELECT

-- Sab columns
SELECT * FROM students;

-- Specific columns
SELECT name, email, marks FROM students;

-- With aliases
SELECT name AS student_name, marks AS score FROM students;

-- DISTINCT (unique values)
SELECT DISTINCT grade FROM students;

-- COUNT total rows
SELECT COUNT(*) FROM students;
SELECT COUNT(DISTINCT grade) FROM students;

SELECT with Expressions

-- Calculated column
SELECT name, marks, marks * 0.4 AS internal_marks FROM students;

-- Concatenation (MySQL/PostgreSQL)
SELECT CONCAT(name, ' - ', grade) AS info FROM students;

-- String functions
SELECT UPPER(name) FROM students;
SELECT LOWER(email) FROM students;
SELECT LENGTH(name) FROM students;

🎯 WHERE & Conditions

Comparison Operators

SELECT * FROM students WHERE age > 20;
SELECT * FROM students WHERE marks >= 80;
SELECT * FROM students WHERE grade = 'A';
SELECT * FROM students WHERE grade != 'C';  -- ya <>
SELECT * FROM students WHERE age BETWEEN 18 AND 22;
SELECT * FROM students WHERE name LIKE 'A%';  -- 'A' se start
SELECT * FROM students WHERE email LIKE '%@example.com';

Logical Operators

-- AND (dono conditions true)
SELECT * FROM students 
WHERE marks >= 80 AND grade = 'A';

-- OR (koi ek true)
SELECT * FROM students 
WHERE grade = 'A' OR grade = 'B';

-- NOT (reverse)
SELECT * FROM students WHERE NOT grade = 'C';

-- Combined
SELECT * FROM students 
WHERE (marks >= 80 AND grade = 'A') OR age < 20;

IN & NOT IN

SELECT * FROM students WHERE grade IN ('A', 'B');
SELECT * FROM students WHERE grade NOT IN ('C', 'D');

IS NULL & IS NOT NULL

SELECT * FROM students WHERE email IS NULL;
SELECT * FROM students WHERE email IS NOT NULL;

✏️ UPDATE & DELETE

UPDATE

-- Single column
UPDATE students SET grade = 'A' WHERE id = 1;

-- Multiple columns
UPDATE students 
SET grade = 'B', marks = 75.50 
WHERE id = 2;

-- All rows (BE CAREFUL!)
UPDATE students SET is_active = TRUE;

-- Increment value
UPDATE students SET marks = marks + 5 WHERE grade = 'B';
⚠️ Hamesha WHERE clause use karo UPDATE/DELETE mein! Bina WHERE ke sab rows update/delete ho jayengi!

DELETE

-- Specific row
DELETE FROM students WHERE id = 5;

-- Multiple rows
DELETE FROM students WHERE grade = 'F';

-- All rows (table empty ho jayegi, structure rahegi)
DELETE FROM students;

-- Table hi delete karna hai
DROP TABLE students;

TRUNCATE (Fast Delete All)

TRUNCATE TABLE students;  -- Sab rows delete, auto-increment reset

📊 ORDER BY & LIMIT

ORDER BY

-- Ascending (default)
SELECT * FROM students ORDER BY marks;
SELECT * FROM students ORDER BY marks ASC;

-- Descending
SELECT * FROM students ORDER BY marks DESC;

-- Multiple columns
SELECT * FROM students ORDER BY grade ASC, marks DESC;

-- By position
SELECT name, marks FROM students ORDER BY 2 DESC; -- 2nd column

LIMIT / OFFSET (Pagination)

-- Top 5 students (MySQL/PostgreSQL/SQLite)
SELECT * FROM students ORDER BY marks DESC LIMIT 5;

-- Skip 10, show next 5
SELECT * FROM students ORDER BY marks DESC LIMIT 5 OFFSET 10;

-- Alternative syntax (SQL Server)
SELECT TOP 5 * FROM students ORDER BY marks DESC;

-- Alternative (MySQL shorthand)
SELECT * FROM students LIMIT 10, 5; -- OFFSET 10, LIMIT 5

📈 Aggregate Functions

Aggregate functions multiple rows par calculation karte hain aur single result dete hain.

-- COUNT
SELECT COUNT(*) FROM students;
SELECT COUNT(*) FROM students WHERE grade = 'A';
SELECT COUNT(DISTINCT grade) FROM students;

-- SUM
SELECT SUM(marks) FROM students;
SELECT SUM(salary) FROM employees WHERE dept_id = 1;

-- AVG (Average)
SELECT AVG(marks) FROM students;
SELECT ROUND(AVG(marks), 2) FROM students; -- 2 decimal places

-- MIN & MAX
SELECT MIN(marks) FROM students;
SELECT MAX(marks) FROM students;
SELECT name, MAX(marks) FROM students; -- Error! Group by chahiye

-- Combined
SELECT 
    COUNT(*) AS total_students,
    AVG(marks) AS average_marks,
    MIN(marks) AS lowest,
    MAX(marks) AS highest,
    SUM(marks) AS total_marks
FROM students;

📦 GROUP BY & HAVING

GROUP BY

GROUP BY rows ko groups mein divide karta hai aur har group par aggregate function lagata hai.

-- Grade wise count
SELECT grade, COUNT(*) AS count FROM students GROUP BY grade;

-- Grade wise average marks
SELECT grade, AVG(marks) AS avg_marks 
FROM students 
GROUP BY grade;

-- Department wise stats
SELECT 
    d.dept_name,
    COUNT(e.emp_id) AS emp_count,
    AVG(e.salary) AS avg_salary,
    MAX(e.salary) AS max_salary
FROM employees e
JOIN departments d ON e.dept_id = d.dept_id
GROUP BY d.dept_name;

HAVING

HAVING GROUP BY ke baad filter karta hai (WHERE row-level filter karta hai).

-- Groups having more than 5 students
SELECT grade, COUNT(*) AS count 
FROM students 
GROUP BY grade 
HAVING COUNT(*) > 5;

-- Average marks > 80 wale grades
SELECT grade, AVG(marks) AS avg_marks 
FROM students 
GROUP BY grade 
HAVING AVG(marks) > 80;

-- WHERE + GROUP BY + HAVING
SELECT grade, COUNT(*) AS count, AVG(marks) AS avg
FROM students 
WHERE age >= 18
GROUP BY grade 
HAVING COUNT(*) >= 3
ORDER BY avg DESC;
WHERE vs HAVING: WHERE row-level filter hai (GROUP BY se pehle), HAVING group-level filter hai (GROUP BY ke baad).

🔗 JOINs

JOIN multiple tables se data ek saath fetch karne ka tareeka hai.

Types of JOINs

JOINResult
INNER JOINMatching rows dono tables se
LEFT JOINLeft table sab + Right se matching
RIGHT JOINRight table sab + Left se matching
FULL OUTER JOINSab rows dono tables se
CROSS JOINCartesian product (sab combinations)
SELF JOINTable khud se join

INNER JOIN

SELECT e.name, d.dept_name, e.salary
FROM employees e
INNER JOIN departments d ON e.dept_id = d.dept_id;

-- Multiple JOINs
SELECT 
    e.name AS employee,
    d.dept_name AS department,
    p.project_name
FROM employees e
JOIN departments d ON e.dept_id = d.dept_id
JOIN projects p ON e.emp_id = p.emp_id;

LEFT JOIN

-- Sab employees + unke departments (bina dept wale bhi)
SELECT e.name, d.dept_name
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.dept_id;

-- Employees without department
SELECT e.name
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.dept_id
WHERE d.dept_id IS NULL;

RIGHT JOIN

-- Sab departments + unke employees (empty dept bhi)
SELECT d.dept_name, e.name
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.dept_id;

FULL OUTER JOIN

-- Sab employees aur sab departments
SELECT e.name, d.dept_name
FROM employees e
FULL OUTER JOIN departments d ON e.dept_id = d.dept_id;

SELF JOIN

-- Employee aur unka manager (same table)
SELECT 
    e.name AS employee,
    m.name AS manager
FROM employees e
JOIN employees m ON e.manager_id = m.emp_id;

CROSS JOIN

-- Sab combinations
SELECT s.name, c.course_name
FROM students s
CROSS JOIN courses c;

🔧 ALTER & DROP

ALTER TABLE

-- Column add karna
ALTER TABLE students ADD phone VARCHAR(15);

-- Column delete karna
ALTER TABLE students DROP COLUMN phone;

-- Column modify karna
ALTER TABLE students MODIFY COLUMN age SMALLINT;
ALTER TABLE students ALTER COLUMN marks TYPE DECIMAL(6,2);

-- Column rename karna
ALTER TABLE students RENAME COLUMN email TO email_address;

-- Constraint add karna
ALTER TABLE students ADD CONSTRAINT chk_age CHECK (age >= 0);

-- Constraint delete karna
ALTER TABLE students DROP CONSTRAINT chk_age;

-- Default value change
ALTER TABLE students ALTER COLUMN is_active SET DEFAULT FALSE;

-- Column ko NOT NULL banana
ALTER TABLE students MODIFY name VARCHAR(100) NOT NULL;

DROP

-- Table delete
DROP TABLE students;

-- Table delete (agar exist karti hai)
DROP TABLE IF EXISTS students;

-- Database delete
DROP DATABASE school_db;

-- Database create
CREATE DATABASE school_db;
USE school_db;

🔍 Subqueries

Subquery ek query ke andar dusri query hai.

Subquery in WHERE

-- Average marks se zyada wale students
SELECT * FROM students 
WHERE marks > (SELECT AVG(marks) FROM students);

-- Same department mein sabse zyada salary wale employee
SELECT * FROM employees e
WHERE salary = (
    SELECT MAX(salary) 
    FROM employees 
    WHERE dept_id = e.dept_id
);

Subquery in FROM

-- Derived table
SELECT * FROM (
    SELECT grade, AVG(marks) AS avg_marks 
    FROM students 
    GROUP BY grade
) AS grade_stats
WHERE avg_marks > 80;

Subquery in SELECT

SELECT 
    name,
    marks,
    (SELECT AVG(marks) FROM students) AS class_avg,
    marks - (SELECT AVG(marks) FROM students) AS difference
FROM students;

EXISTS & NOT EXISTS

-- Students jinke orders hain
SELECT * FROM customers c
WHERE EXISTS (
    SELECT 1 FROM orders o WHERE o.customer_id = c.customer_id
);

-- Students jinke orders nahi hain
SELECT * FROM customers c
WHERE NOT EXISTS (
    SELECT 1 FROM orders o WHERE o.customer_id = c.customer_id
);

IN & NOT IN with Subquery

-- Departments jinme employees hain
SELECT * FROM departments 
WHERE dept_id IN (SELECT DISTINCT dept_id FROM employees);

-- Departments jinme employees nahi hain
SELECT * FROM departments 
WHERE dept_id NOT IN (SELECT DISTINCT dept_id FROM employees WHERE dept_id IS NOT NULL);

⚡ Indexes

Indexes search speed ko improve karte hain. Book ka index jaisa — direct page number mil jata hai.

Creating Indexes

-- Single column index
CREATE INDEX idx_name ON students(name);

-- Unique index
CREATE UNIQUE INDEX idx_email ON students(email);

-- Composite index (multiple columns)
CREATE INDEX idx_name_age ON students(name, age);

-- Full-text index (MySQL)
CREATE FULLTEXT INDEX idx_content ON articles(content);

-- Index delete karna
DROP INDEX idx_name ON students;

When to Use

  • Columns jinme frequently WHERE, JOIN, ORDER BY use hota hai
  • Columns jinme UNIQUE values hain
  • Large tables (small tables mein index slow kar sakta hai)

When NOT to Use

  • Small tables
  • Columns jinme frequently INSERT/UPDATE hota hai
  • Columns jinme mostly NULL values hain

🛠️ Mini Projects

1. School Database Schema

-- Students, Teachers, Subjects, Marks

CREATE TABLE students (
    student_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    class VARCHAR(10),
    roll_no INT,
    dob DATE,
    address TEXT
);

CREATE TABLE subjects (
    subject_id INT PRIMARY KEY AUTO_INCREMENT,
    subject_name VARCHAR(50) NOT NULL,
    max_marks INT DEFAULT 100
);

CREATE TABLE marks (
    mark_id INT PRIMARY KEY AUTO_INCREMENT,
    student_id INT,
    subject_id INT,
    marks_obtained DECIMAL(5,2),
    exam_date DATE,
    FOREIGN KEY (student_id) REFERENCES students(student_id),
    FOREIGN KEY (subject_id) REFERENCES subjects(subject_id)
);

-- Top 10 students
SELECT s.name, SUM(m.marks_obtained) AS total_marks
FROM students s
JOIN marks m ON s.student_id = m.student_id
GROUP BY s.student_id, s.name
ORDER BY total_marks DESC
LIMIT 10;

-- Subject wise average
SELECT sub.subject_name, AVG(m.marks_obtained) AS avg_marks
FROM subjects sub
JOIN marks m ON sub.subject_id = m.subject_id
GROUP BY sub.subject_id, sub.subject_name;

2. E-Commerce Database

CREATE TABLE customers (
    customer_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    email VARCHAR(100) UNIQUE,
    phone VARCHAR(15),
    city VARCHAR(50)
);

CREATE TABLE products (
    product_id INT PRIMARY KEY AUTO_INCREMENT,
    product_name VARCHAR(100),
    price DECIMAL(10,2),
    stock INT DEFAULT 0,
    category VARCHAR(50)
);

CREATE TABLE orders (
    order_id INT PRIMARY KEY AUTO_INCREMENT,
    customer_id INT,
    order_date DATE DEFAULT CURRENT_DATE,
    total DECIMAL(10,2),
    status VARCHAR(20) DEFAULT 'pending',
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id)
);

CREATE TABLE order_items (
    item_id INT PRIMARY KEY AUTO_INCREMENT,
    order_id INT,
    product_id INT,
    quantity INT,
    price DECIMAL(10,2),
    FOREIGN KEY (order_id) REFERENCES orders(order_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
);

-- Total sales by month
SELECT 
    DATE_FORMAT(order_date, '%Y-%m') AS month,
    COUNT(*) AS total_orders,
    SUM(total) AS revenue
FROM orders
WHERE status = 'delivered'
GROUP BY month
ORDER BY month;

3. Library Management System

CREATE TABLE books (
    book_id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(200),
    author VARCHAR(100),
    isbn VARCHAR(20) UNIQUE,
    category VARCHAR(50),
    available BOOLEAN DEFAULT TRUE
);

CREATE TABLE members (
    member_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    email VARCHAR(100) UNIQUE,
    join_date DATE DEFAULT CURRENT_DATE
);

CREATE TABLE borrowings (
    borrow_id INT PRIMARY KEY AUTO_INCREMENT,
    book_id INT,
    member_id INT,
    borrow_date DATE,
    due_date DATE,
    return_date DATE,
    fine DECIMAL(6,2) DEFAULT 0,
    FOREIGN KEY (book_id) REFERENCES books(book_id),
    FOREIGN KEY (member_id) REFERENCES members(member_id)
);

-- Overdue books
SELECT b.title, m.name, br.due_date,
    DATEDIFF(CURRENT_DATE, br.due_date) AS days_overdue
FROM borrowings br
JOIN books b ON br.book_id = b.book_id
JOIN members m ON br.member_id = m.member_id
WHERE br.return_date IS NULL 
AND br.due_date < CURRENT_DATE;

-- Most popular books
SELECT b.title, COUNT(*) AS times_borrowed
FROM books b
JOIN borrowings br ON b.book_id = br.book_id
GROUP BY b.book_id, b.title
ORDER BY times_borrowed DESC
LIMIT 10;

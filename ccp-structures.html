C++ Structures Complete Guide

Creating a Structure

struct Student {
    string name;
    int age;
    float marks;
};

Student s1;
s1.name = "Akarshit";
s1.age = 18;
s1.marks = 95.5;

cout << s1.name << " " << s1.age;

Initialization

Student s2 = {"Rahul", 17, 88.0};
Student s3 = {"Priya", 18, 92.5};

Array of Structures

Student class10[3] = {
    {"Akarshit", 18, 95},
    {"Rahul", 17, 88},
    {"Priya", 18, 92}
};

for (int i = 0; i < 3; i++) {
    cout << class10[i].name << endl;
}

Nested Structure

struct Address {
    string city;
    int pin;
};

struct Person {
    string name;
    Address addr;
};

Person p = {"Akarshit", {"Sitapur", 261001}};
cout << p.addr.city;  // Sitapur

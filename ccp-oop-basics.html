C++ OOP Basics Complete Guide

Class & Object

class Student {
public:
    string name;
    int age;
    
    void display() {
        cout << name << " - " << age;
    }
};

int main() {
    Student s1;
    s1.name = "Akarshit";
    s1.age = 18;
    s1.display();
}

Constructor

class Student {
public:
    string name;
    int age;
    
    Student(string n, int a) {
        name = n;
        age = a;
    }
};

Student s1("Akarshit", 18);
s1.display();

Encapsulation (Private)

class BankAccount {
private:
    double balance;
    
public:
    void deposit(double amount) {
        balance += amount;
    }
    
    double getBalance() {
        return balance;
    }
};

Inheritance

class Animal {
public:
    void eat() {
        cout << "Eating...";
    }
};

class Dog : public Animal {
public:
    void bark() {
        cout << "Barking...";
    }
};

Dog d;
d.eat();   // Inherited
d.bark();  // Own method

Polymorphism

class Shape {
public:
    virtual void draw() {
        cout << "Drawing shape";
    }
};

class Circle : public Shape {
public:
    void draw() override {
        cout << "Drawing circle";
    }
};

Shape* s = new Circle();
s->draw();  // Drawing circle

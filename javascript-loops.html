JavaScript Loops Complete Guide

For Loop

for (let i = 0; i < 5; i++) {
    console.log(i);
}
// Output: 0 1 2 3 4

While Loop

let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}

Do-While Loop

let i = 0;
do {
    console.log(i);
    i++;
} while (i < 5);

For...Of (Arrays)

const fruits = ["apple", "banana", "cherry"];
for (let fruit of fruits) {
    console.log(fruit);
}
// Output: apple banana cherry

For...In (Objects)

const student = { name: "Akarshit", age: 18 };
for (let key in student) {
    console.log(key + ": " + student[key]);
}
// Output: name: Akarshit, age: 18

forEach Method

const numbers = [1, 2, 3, 4, 5];
numbers.forEach((num) => {
    console.log(num * 2);
});
// Output: 2 4 6 8 10

Break & Continue

for (let i = 0; i < 10; i++) {
    if (i === 3) continue;  // skip 3
    if (i === 7) break;     // stop at 7
    console.log(i);
}
// Output: 0 1 2 4 5 6

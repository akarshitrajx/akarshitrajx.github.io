TypeScript Complete Tutorial

What is TypeScript?

TypeScript is JavaScript with added types. It catches errors at compile time and makes code more maintainable.

Basic Types

let name: string = "Akarshit";
let age: number = 18;
let isActive: boolean = true;
let hobbies: string[] = ["coding", "gaming"];
let user: [string, number] = ["Akarshit", 18];  // tuple

Interfaces

interface Student {
    name: string;
    age: number;
    grade?: string;  // optional
}

const s1: Student = {
    name: "Akarshit",
    age: 18,
    grade: "A"
};

Functions with Types

function add(a: number, b: number): number {
    return a + b;
}

const greet = (name: string): string => {
    return `Hello, ${name}`;
};

Classes

class Animal {
    private name: string;
    
    constructor(n: string) {
        this.name = n;
    }
    
    move(distance: number): void {
        console.log(`${this.name} moved ${distance}m`);
    }
}

const dog = new Animal("Dog");
dog.move(10);

C++ Arrays Complete Guide

1D Array

int marks[5] = {85, 90, 78, 92, 88};
cout << marks[0];  // 85
marks[0] = 95;     // modify

// Loop through array
for (int i = 0; i < 5; i++) {
    cout << marks[i] << " ";
}

2D Array

int matrix[2][3] = {
    {1, 2, 3},
    {4, 5, 6}
};
cout << matrix[1][2];  // 6

Array Size

int arr[] = {10, 20, 30, 40};
int size = sizeof(arr) / sizeof(arr[0]);
cout << size;  // 4

Array as Function Parameter

void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
}

int nums[] = {1, 2, 3, 4, 5};
printArray(nums, 5);

C++ Loops Complete Guide

For Loop

for (int i = 0; i < 5; i++) {
    cout << i << " ";
}
// Output: 0 1 2 3 4

While Loop

int i = 0;
while (i < 5) {
    cout << i << " ";
    i++;
}

Do-While Loop

int i = 0;
do {
    cout << i << " ";
    i++;
} while (i < 5);

Break & Continue

for (int i = 0; i < 10; i++) {
    if (i == 3) continue;  // skip 3
    if (i == 7) break;     // stop at 7
    cout << i << " ";
}
// Output: 0 1 2 4 5 6

Range-based For (C++11)

int arr[] = {10, 20, 30};
for (int x : arr) {
    cout << x << " ";
}
// Output: 10 20 30

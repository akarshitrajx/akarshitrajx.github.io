Type Casting

int a = 10;
double b = (double)a / 3;  // 3.333...
int c = (int)3.99;          // 3 (truncated)

auto Keyword (C++11)

auto x = 10;        // int
auto y = 3.14;      // double
auto z = "Hello";   // const char*
